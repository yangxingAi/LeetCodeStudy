public class Offer55 {
    public int maxDepth(TreeNode root) {
        if(root == null) return 0;
        int leftDep = maxDepth(root.left);
        int rightDep = maxDepth(root.right);
        return leftDep > rightDep ? leftDep++ : rightDep++;
    }
}

 class TreeNode {
      int val;
      TreeNode left;
     TreeNode right;
     TreeNode(int x) { val = x; }
}
